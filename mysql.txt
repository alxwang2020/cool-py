mysql用法 以数据库nad > 表class为例
1.查看数据库
show databases;
2.新建数据库
creat database nad;
3.删除数据库
drop database nad;
4.调用
use nad;
5.新建表格
create table class values('abc',3,5),();
6.查看表格内容
select * from class;
   6.1 select name,age from class;
   6.2
    select * from class order by AGE desc降序/asc升序;
7.查看表格属性
desc class;
8.插入数据
insert into class values('abc','1',3),(),();
9.删除列
delete from class where NAME="abc";
10.更改数据
    10.1单个修改
update class set AGE=12 where ID="";
    10.2使用case when多个修改
update class
set NAME = case AGE
when 1 then "NAME值"
when "7" then ""
...
end 
where AGE in (1,7)
